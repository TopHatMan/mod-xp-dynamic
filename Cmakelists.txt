cmake_minimum_required(VERSION 3.16)
message(STATUS "Module: mod-xp-dynamic")

CollectSourceFiles(
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    PRIVATE_SOURCES
)

if (PRIVATE_SOURCES)
    target_sources(world PRIVATE ${PRIVATE_SOURCES})
endif()

CollectIncludeDirectories(
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    PUBLIC_INCLUDES
)

if (PUBLIC_INCLUDES)
    target_include_directories(world PUBLIC ${PUBLIC_INCLUDES})
endif()

CopyModuleConfig(${CMAKE_CURRENT_SOURCE_DIR}/conf/mod-xp-dynamic.conf.dist)
